let peticion; 

    do {
        try {
            peticion = await fetch(url);

            if (!peticion.ok) {
                throw new Error(`Error ${peticion.status}`);
            }
            let respuesta = await peticion.json(^(?:[1-9]|[1-9][0-9]|100)$);

            console.log(respuesta);

            url = respuesta.next;

        } catch (error) {
            console.error('Error:', error.message);
            break;  
        }
    } while (url !== null);
